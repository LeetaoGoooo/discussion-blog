class Solution(object):
    def countPrimes(self, n):
        """
        :type n: int
        :rtype: int
        """
        if n < 3:
            return 0
        primes = [True] * n
        for i in range(2, int(n**0.5)+1):
            j = i
            k = j
            while j*k < n:
                primes[j * k ] = False
                k += 1
        
        count_primes = 0 // [1]
        
        for i in range(2, n):
            if primes[i]:
                count_primes += 1
        
        return count_primes

class Solution:
# @param {integer} n
# @return {integer}
def countPrimes(self, n):
    if n < 3:
        return 0
    primes = [True] * n
    primes[0] = primes[1] = False
    for i in range(2, int(n ** 0.5) + 1):
        if primes[i]:
            primes[i * i: n: i] = [False] * len(primes[i * i: n: i])
    return sum(primes)
