use std::collections::HashMap;

impl Solution {
    pub fn majority_element(nums: Vec<i32>) -> i32 {
        let mut numsMap = HashMap::new();
        for num in nums {
            let count = numsMap.entry(num).or_insert(0);
            *count += 1
        }
        let mut maxValue = 0;
        let mut maxKey = 0;
        for(key, value) in numsMap {
            if value > maxValue {
                maxKey = key;
                maxValue = value;
            }
        }
        return maxKey;
    }
}

impl Solution {
    pub fn majority_element(nums: Vec<i32>) -> i32 {

        let mut count = 0;
        for i in 0..nums.len()
        {
            if(count == 0)
            {
                num = nums[i];
                count = 1;
            }
            else if nums[i] != num
            {
                count-=1;
            }
            else
            {
                count+=1;
            }
        }
